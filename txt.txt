
When  .navlink-language is clicked


if page language = ja 
    if url = <domain>index.html 
    redirect to en.html

    else if url = <domain><extension>
    inset /en/ after <domain>
    redirect to <domain>/en/<extension>

    end if

elseif page language = en
    if url = <domain>en.html
    redirect to index.html

    else if = <domain>/en/<extension>
    remove /en/ after <domain>
    redirect to <domain><extension>

end if
